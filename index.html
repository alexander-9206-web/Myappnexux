        function showLoanDetail(id) {

            const loan = state.loans.find(l => l.id === id);

            if(!loan) return;


            document.getElementById('loans-view-list').classList.add('hidden');

            document.getElementById('loans-view-detail').classList.remove('hidden');

            document.getElementById('det-loan-title').innerText = loan.name;


            // Variables del Método Francés

            const P = loan.total;

            const r = loan.rate / 100; // Tasa Mensual

            const n = loan.term;

            

            // Cuota Fija (A)

            const A = (r === 0) ? P / n : P * (r * Math.pow(1 + r, n)) / (Math.pow(1 + r, n) - 1);


            // Calcular meses transcurridos desde el desembolso

            const startDate = new Date(loan.date);

            const now = new Date();

            let monthsPassed = (now.getFullYear() - startDate.getFullYear()) * 12 + (now.getMonth() - startDate.getMonth());

            if(monthsPassed < 0) monthsPassed = 0;

            if(monthsPassed > n) monthsPassed = n;


            let balance = P;

            let paidToDate = 0;

            let pendingCapital = P;

            let totalIntPending = 0;

            let totalIntAcc = 0;

            

            let htmlRows = '';


            for (let i = 1; i <= n; i++) {

                const interest = balance * r;

                const capital = A - interest;

                totalIntAcc += interest;

                

                // Formatear Fecha

                const pDate = new Date(startDate);

                pDate.setMonth(pDate.getMonth() + i);

                const dateStr = pDate.toLocaleDateString('es-CO', {month:'short', year:'2-digit'});


                // Lógica de "Pagado" vs "Pendiente"

                let rowClass = "";

                let statusIcon = "";

                if (i <= monthsPassed) {

                    paidToDate += A;

                    pendingCapital -= capital;

                    rowClass = "opacity-40 line-through text-slate-400";

                    statusIcon = `<i class="fa-solid fa-check text-emerald-500 mr-2"></i>`;

                } else {

                    totalIntPending += interest;

                    rowClass = "font-medium";

                    statusIcon = `<span class="mr-2 text-slate-300">${i}</span>`;

                }


                balance -= capital;


                htmlRows += `

                    <tr class="border-b border-slate-50 ${rowClass}">

                        <td class="px-4 py-4 uppercase flex items-center">${statusIcon}${dateStr}</td>

                        <td class="px-4 py-4 text-right font-mono">${fmt.format(capital)}</td>

                        <td class="px-4 py-4 text-right font-mono">${fmt.format(interest)}</td>

                        <td class="px-4 py-4 text-right font-black">${fmt.format(A)}</td>

                        <td class="px-4 py-4 text-right font-mono text-amber-600">${fmt.format(Math.max(0, balance))}</td>

                    </tr>

                `;

            }


            document.getElementById('det-loan-table').innerHTML = htmlRows;


            // Generar Párrafo Dinámico

            const summaryHTML = `

                El crédito <b>${loan.name}</b> tiene un valor inicial de <b>${fmt.format(P)}</b> diferido a <b>${n} meses</b> con una tasa del <b>${loan.rate}% E.M</b>. 

                A la fecha, se han cubierto ${monthsPassed} periodos por un valor pagado de <b>${fmt.format(paidToDate)}</b>, dejando un capital pendiente de <b>${fmt.format(pendingCapital)}</b>. 

                El valor de intereses futuros a pagar es de <b>${fmt.format(totalIntPending)}</b>.

                <br><br>

                💡 <span class="uppercase font-black tracking-widest text-[9px] text-amber-600">Tip Financiero</span><br>

                Realizar abonos a capital reducirá el interés futuro de ${fmt.format(totalIntPending)}. Puedes elegir entre <b>reducir el plazo</b> (ahorras más intereses) o <b>reducir la cuota mensual</b> (mejora tu flujo de caja), siendo la reducción en tiempo la opción más rentable.

            `;

            document.getElementById('det-loan-paragraph').innerHTML = summaryHTML;

        }

        function getNextPayDate(baseDate, targetDayOfWeek) {

            let date = new Date(baseDate);

            let currentDay = date.getDay();

            let distance = targetDayOfWeek - currentDay;

            if (distance <= 0) distance += 7;

            return date;

        }


        function calculateAfterpaySchedules() {

            const apTx = state.tx.filter(t => t.type === 'afterpay' && (t.apPaid || 0) < 4);

            const globalSchedule = [];

        function renderReport() {

            const now = new Date();

            const last90Days = new Date(now.getTime() - 90 * 86400000);

            const last30Days = new Date(now.getTime() - 30 * 86400000);


            // 1. Cálculos de Activos y Pasivos

            const totalAssets = state.accounts.filter(a => a.type !== 'credit').reduce((s,a) => s + a.balance, 0);

            

            // Pasivos: Tarjetas de Crédito + Préstamos Bancarios + Afterpay

            const creditCardDebt = state.accounts.filter(a => a.type === 'credit').reduce((s,a) => s + a.balance, 0);

            const activeLoans = state.loans.filter(l => !l.isBNPL);

            let loanDebt = 0;

            let monthlyDebtService = 0;

            activeLoans.forEach(l => {

                const P = l.total;

                const r = l.rate / 100;

                const n = l.term;

                const A = (r === 0) ? P / n : P * (r * Math.pow(1 + r, n)) / (Math.pow(1 + r, n) - 1);

                monthlyDebtService += A;


                const startDate = new Date(l.date);

                let monthsPassed = (now.getFullYear() - startDate.getFullYear()) * 12 + (now.getMonth() - startDate.getMonth());

                if(monthsPassed < 0) monthsPassed = 0;

                if(monthsPassed > n) monthsPassed = n;

                

                let balance = P;

                for (let i = 1; i <= monthsPassed; i++) {

                    const interest = balance * r;

                    balance -= (A - interest);

                }

                loanDebt += Math.max(0, balance);

            });


            const apTx = state.tx.filter(t => t.type === 'afterpay' && (t.apPaid || 0) < 4);

            const pendingAp = apTx.reduce((sum, tx) => sum + (tx.amount * (4 - (tx.apPaid||0)) / 4), 0);


            const totalLiabilities = Math.abs(creditCardDebt) + loanDebt + pendingAp;


            // 2. Cálculo Operativo (Últimos 90 Días para Tasa de Retención)

            const recentTx = state.tx.filter(t => new Date(t.date) >= last90Days && t.type !== 'afterpay');

            const recentInc = recentTx.filter(t => t.type === 'income').reduce((s,t) => s + t.amount, 0);

            const recentExp = recentTx.filter(t => t.type === 'expense').reduce((s,t) => s + t.amount, 0);

            

            const retentionRate = recentInc > 0 ? (((recentInc - recentExp) / recentInc) * 100).toFixed(1) : 0;

            const monthlyIncAvg = recentInc / 3;


            // 3. Ratio Deuda/Ingreso

            const debtToIncome = monthlyIncAvg > 0 ? ((monthlyDebtService / monthlyIncAvg) * 100).toFixed(1) : 0;


            // DOM KPIs

            document.getElementById('rep-assets').innerText = fmt.format(totalAssets);

            document.getElementById('rep-liabilities').innerText = fmt.format(totalLiabilities);

                const pastMonthsCount = currentMonth + 1;

                const pastExp = totalExp / pastMonthsCount;

                        mInc = mTx.filter(t => t.type === 'income').reduce((s,t) => s+t.amount, 0);

                        mInc = pastInc; mExp = pastExp; isProj = true;

                    cumulativeNet += mNet;

                        </tr>

                    `;

                }

                document.getElementById('stat-year-tbody').innerHTML = yearHTML;

            }

        function saveTransaction(e) {

            e.preventDefault();

            const type = document.getElementById('f-type').value;

            const amount = parseFloat(document.getElementById('f-amount').value);

            const cat = document.getElementById('f-cat').value;

            const accId = document.getElementById('f-account').value;

            const date = document.getElementById('f-date').value;

            const note = document.getElementById('f-note').value || cat;


            if(!amount || isNaN(amount)) return alert("Monto inválido");


            const apFreq = type === 'afterpay' ? document.getElementById('f-ap-freq').value : null;


            state.tx.push({

                id: Date.now().toString(),

                type, amount, cat, 

                accId: type === 'afterpay' ? 'afterpay_credit' : accId, 

                date: new Date(date).toISOString(), 

                note,

                ...(type === 'afterpay' && { apFreq, apPaid: 0 })

            });


            if(type !== 'afterpay') {

                const acc = state.accounts.find(a => a.id === accId);

                if(acc) {

                    if(type === 'income') acc.balance += amount;

                    else acc.balance -= amount;

                }

            }


            saveState();

            closeForm();

            updateDashboard();

            if(state.activeTab === 'analysis') renderAnalysis(); 

            if(state.activeTab === 'afterpay') renderAfterpay(); 

        }

        function exportBackup() {

            try {

                const dataStr = JSON.stringify(state, null, 2);

                const blob = new Blob([dataStr], { type: "application/json" });

                const url = URL.createObjectURL(blob);

                const a = document.createElement('a');

                a.href = url;

                a.download = `Nexus_Backup_${new Date().toISOString().split('T')[0]}.json`;

                document.body.appendChild(a);

                a.click();

                document.body.removeChild(a);

                URL.revokeObjectURL(url);

            } catch (err) {

                alert("Error al exportar el respaldo.");

            }

        }


        function importBackup(event) {

            const file = event.target.files[0];

            if (!file) return;

            const reader = new FileReader();

            reader.onload = function(e) {

                try {

                    const importedState = JSON.parse(e.target.result);

                    if (importedState && importedState.tx && Array.isArray(importedState.tx)) {

                        state = importedState;

                        saveState();

                        alert("✅ Respaldo importado correctamente. El sistema se reiniciará.");

                        location.reload();

                    } else {

                        alert("❌ Error: Archivo incorrecto.");

                    }

                } catch (err) {

                    alert("❌ Error al leer el archivo JSON.");

                }

            };

            reader.readAsText(file);

        }
